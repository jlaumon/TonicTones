00061 void ImageScrollArea::wheelEvent(QWheelEvent *event)
00062 {
00063     int numDegrees = event->delta() / 8;
00064     int numSteps = numDegrees / 15;
00065     
00066     if (event->modifiers() & Qt::ShiftModifier)
00067         horizontalScrollBar()->setValue(horizontalScrollBar()->value() - 3*numSteps*horizontalScrollBar()->singleStep());
00068     else if (event->modifiers() & Qt::ControlModifier)
00069         scaleImage(1.0+(double(numSteps)/10.0));
00070     else 
00071         verticalScrollBar()->setValue(verticalScrollBar()->value() - 3*numSteps*verticalScrollBar()->singleStep());
00072     
00073     event->accept();
00074 }
